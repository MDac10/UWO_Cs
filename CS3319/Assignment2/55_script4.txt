USE 55_assign2db;

-- Create view
CREATE VIEW tripInfo AS SELECT DISTINCT firstName, lastName, tripName, country, price FROM passenger, bustrip, booking WHERE passenger.passengerID = booking.passengerID AND bustrip.tripID = booking.tripID;

-- View works
SELECT * FROM tripInfo;

-- Looking for Castles
SELECT * FROM tripInfo WHERE tripName LIKE '%Castles%';
